SET search_path TO atividadesql2;

-- PROCEDURE 1: quadro_de_medalhas
-- Objetivo: Contabilizar Ouro, Prata e Bronze por clube
CREATE OR REPLACE PROCEDURE quadro_de_medalhas()
LANGUAGE plpgsql
AS $$
BEGIN
   -- Esta consulta agrupa os resultados para contar medalhas
   -- ALUCINAÇÃO POTENCIAL: Como a IA define quem ganhou ouro/prata/bronze?
   SELECT
      c.nome_clube,
      COUNT(*) FILTER (WHERE r.colocacao = 1) AS ouro,
      COUNT(*) FILTER (WHERE r.colocacao = 2) AS prata,
      COUNT(*) FILTER (WHERE r.colocacao = 3) AS bronze
   FROM atividadesql2.CLUBE c
   JOIN atividadesql2.ATLETA a ON c.id_clube = a.id_clube
   JOIN atividadesql2.RESULTADO r ON a.id_atleta = r.id_atleta
   GROUP BY c.nome_clube
   ORDER BY ouro DESC, prata DESC, bronze DESC;
END;
$$;

-- Objetivo: Desempenho por equipe e filtragem por categoria

    SELECT

    GROUP BY a.nome_atleta, c.nome_clube, a.data_nascimento;
  END;
  $$;

  Estratégia de Normalização (3ª Forma Normal) A principal decisão de design foi a
  decomposição de atributos de texto em tabelas de domínio (Lookup Tables).
  Entidades de Domínio (SEXO, ESTADO, ESTILO, CATEGORIA): * Decisão: Em vez de
  armazenar "Masculino" ou "Pernambuco" como strings repetidas nas tabelas de ATLETA
  ou CLUBE, isolamos esses dados.

  Justificativa: Isso elimina a redundância e impede "anomalias de inserção". Garante que
  um estilo de nado seja sempre escrito da mesma forma, facilitando agrupamentos e filtros
  em relatórios de desempenho.
Integridade: O uso de UK (Unique Key) nessas tabelas garante que não existam
duplicatas lógicas (ex: duas entradas para o mesmo estado).

Hierarquia de Localização e Filiação ESTADO (UF) -> CLUBE -> ATLETA:
Decisão: Criamos uma hierarquia clara de dependência.

Justificativa: Um clube está sediado em um estado, e um atleta é filiado a um clube. Ao
separar o ESTADO do CLUBE, permitimos análises macro (ex: ranking de medalhas por
estado) sem precisar varrer e limpar strings de texto nos registros dos clubes.

Especialização da PROVA Composição da PROVA:
Decisão: A entidade PROVA não armazena o nome do estilo ou a categoria diretamente;
ela é uma tabela de interseção técnica que une distancia, id_estilo, id_sexo e
id_categoria.

Justificativa: Isso permite que o sistema gerencie o calendário de eventos de forma
dinâmica. Se a federação criar uma nova categoria (ex: "Master"), basta adicionar um
registro na tabela CATEGORIA e vinculá-lo a novas PROVAS, sem alterar o código do
sistema.

Decomposição de Performance (RESULTADO e PARCIAL) Esta é a área mais crítica do
banco, onde os dados de alta precisão são armazenados.
RESULTADO (Entidade de Agregação):

Decisão: Armazena o desfecho final do atleta em uma prova (tempo total, colocação e
pontos).

Justificativa: Centraliza os dados necessários para o Quadro de Medalhas e Pontuação
de Eficiência visíveis na interface.

PARCIAL (Entidade de Granularidade):

Decisão: Criada como uma entidade dependente (1:N) de RESULTADO.

Justificativa: Conforme solicitado, para suportar a soma de tempos que gera o total, a
tabela PARCIAL permite armazenar múltiplos registros para um único resultado (ex:
parciais de 50m em uma prova de 400m).

Atributo Ordem: O campo ordem é vital para garantir que a sequência dos nados seja
respeitada no cálculo da soma e na exibição do histórico de ritmo do atleta.

Integridade Referencial e Tipagem Relacionamentos Identificadores (--): * Utilizados entre
ATLETA/PROVA -> RESULTADO e RESULTADO -> PARCIAL. Se um atleta for removido
ou uma prova cancelada, os resultados e parciais vinculados perdem o sentido e devem
sofrer exclusão lógica ou física (Cascade).
Relacionamentos Não Identificadores (..): * Utilizados para características biográficas ou
técnicas (Sexo, Categoria, Estilo). O objeto principal (Atleta ou Prova) existe
independentemente de uma alteração na descrição da tabela de domínio.
Tipos de Dados: * Uso de TIME para precisão de milésimos, DECIMAL para pontuações
(evitando erros de arredondamento de ponto flutuante) e INT para chaves primárias
visando performance em indexação.

PROCEDURE 1: quadro_de_medalhas
Esta procedure consolida o desempenho dos clubes. Note que utilizei a tabela RESULTADO como
centralizadora, conforme sua estratégia de "Entidade de Agregação".
SQL
CREATE OR REPLACE PROCEDURE atividadesql2.quadro_de_medalhas()
LANGUAGE plpgsql
AS $$
BEGIN
    -- Relatório de Medalhas por Clube
    -- Considera a colocação (1º, 2º, 3º) para definir Ouro, Prata e Bronze
    DROP TABLE IF EXISTS RelatorioMedalhas; -- Tabela temporária para
visualização do resultado
    CREATE TEMP TABLE RelatorioMedalhas AS
    SELECT
        c.nome_clube,
        c.sigla AS clube_sigla,
        COUNT(r.id_resultado) FILTER (WHERE r.colocacao = 1) AS ouro,
        COUNT(r.id_resultado) FILTER (WHERE r.colocacao = 2) AS prata,
        COUNT(r.id_resultado) FILTER (WHERE r.colocacao = 3) AS bronze,
        SUM(CASE
            WHEN r.colocacao = 1 THEN 9 -- Exemplo de pontuação por eficiência
            WHEN r.colocacao = 2 THEN 7
            WHEN r.colocacao = 3 THEN 6
            ELSE 0
        END) AS pontuacao_total
    FROM atividadesql2.CLUBE c
    JOIN atividadesql2.ATLETA a ON c.id_clube = a.id_clube
    JOIN atividadesql2.RESULTADO r ON a.id_atleta = r.id_atleta
    GROUP BY c.id_clube, c.nome_clube, c.sigla
    ORDER BY ouro DESC, prata DESC, bronze DESC, pontuacao_total DESC;

     -- Nota: Em um ambiente real, poderíamos dar um RAISE NOTICE ou retornar um
Cursor.
     RAISE NOTICE 'Quadro de Medalhas gerado com sucesso na tabela temporária
RelatorioMedalhas.';
END;
$$;




PROCEDURE 2: atletas_e_clubes
Esta procedure foca no desempenho individual e na classificação etária correta, corrigindo as
"alucinações" comuns de IA sobre as categorias da natação brasileira.
SQL
CREATE OR REPLACE PROCEDURE atividadesql2.atletas_e_clubes()
LANGUAGE plpgsql
AS $$
BEGIN
    -- Relatório de Desempenho e Categorização por Idade
    DROP TABLE IF EXISTS RelatorioDesempenhoAtletas;
    CREATE TEMP TABLE RelatorioDesempenhoAtletas AS
    SELECT
        a.nome_atleta,
        c.sigla AS clube,
        -- Lógica Corrigida: Pré-Mirim até 9 anos | Mirim 10 a 11 anos
        CASE
             WHEN (EXTRACT(YEAR FROM CURRENT_DATE) - EXTRACT(YEAR FROM
a.data_nascimento)) <= 9 THEN 'PRÉ-MIRIM'
             WHEN (EXTRACT(YEAR FROM CURRENT_DATE) - EXTRACT(YEAR FROM
a.data_nascimento)) BETWEEN 10 AND 11 THEN 'MIRIM'
             ELSE 'OUTRAS CATEGORIAS'
        END AS categoria_oficial_fase,
        COUNT(r.id_resultado) AS provas_nadadas,
        MIN(r.tempo_final) AS melhor_tempo_registrado -- Assume-se formato de
string comparável ou TIME
    FROM atividadesql2.ATLETA a
    JOIN atividadesql2.CLUBE c ON a.id_clube = c.id_clube
    LEFT JOIN atividadesql2.RESULTADO r ON a.id_atleta = r.id_atleta
    GROUP BY a.id_atleta, a.nome_atleta, c.sigla, a.data_nascimento
    ORDER BY categoria_oficial_fase, a.nome_atleta;

     RAISE NOTICE 'Relatório de Atletas e Clubes (Categorias FASE) gerado em
RelatorioDesempenhoAtletas.';
END;
$$;
